# Setting up sql server environment
driver = 'SQL Server'
server = 'Muchmore\SQLEXPRESS01'
db = 'PIZZA'

# Create connection string
def connection(driver, server, database):
    conn_str = (
        f"DRIVER={{{driver}}};"
        f"SERVER={server};"
        f"DATABASE={database};"
        "Trusted_Connection=yes;"
    )
    return conn_str

# Create database connection object
try:
    conn = odbc.connect(connection(driver, server, db))
    print("Database connection established.")

except odbc.DatabaseError as e:
    print(f"Database Error:\n{str(e)}")
except odbc.Error as e:
    print(f"Connection Error:\n{str(e)}")


# Loading data into SQL
cursor = conn.cursor()

for df, table_name in zip(dfs, names):
    placeholders = ",".join(["?"] * len(df.columns))
    columns = ",".join([f"[{col}]" for col in df.columns])

    sql = f"""
    INSERT INTO {table_name} ({columns})
    VALUES ({placeholders})
    """

    try:
        for _, row in df.iterrows():
            cursor.execute(sql, tuple(row))
        conn.commit()
        print(f"Successfully inserted data into {table_name}")
    except Exception as e:
        print(f"Error inserting data into {table_name}: {str(e)}")
        conn.rollback()
